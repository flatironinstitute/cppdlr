50 template <typename V> bool flip_to_mirror(V const &piv, long mirror_max) {
51 long r = piv.size();
52 for (long i = 0; i < r; ++i) {
53 auto a = piv(i); // original set, ascending
54 auto b = mirror_max - piv(r - 1 - i); // mirror set, ascending
55 if (a != b) return b < a; // first difference decides; flip iff mirror is smaller
56 }
57 return false; // symmetric set: orientations coincide, no flip
58 }

138 template <nda::MemoryArrayOfRank<2> T, nda::Scalar S = nda::get_value_t<T>>
139 std::tuple<typename T::regular_type, nda::vector<double>, nda::vector<int>> pivrgs(T const &a, double eps) {
140
141 // Copy input data
142 auto aa = make_regular(a);
143
144 // Get matrix dimensions
145 auto [m, n] = aa.shape();
146 int maxrnk = std::min(m, n);
147
148 // Compute norms of rows of input matrix, and rescale eps tolerance
149 auto norms = nda::vector<double>(m);
150 double epssq = eps * eps;
151 for (int j = 0; j < m; ++j) { norms(j) = normsq(aa(j, _)); }
152
153 // Begin pivoted double Gram-Schmidt procedure
154 int jpiv = 0;
155 double nrm = 0;
156 auto piv = nda::arange(m);
157 auto tmp = nda::vector<S>(n);
158
159 for (int j = 0; j < maxrnk; ++j) {
160
161 // Find next pivot
162 jpiv = j;
163 nrm = norms(j);
164 for (int k = j + 1; k < m; ++k) {
165 if (norms(k) > nrm) {
166 jpiv = k;
167 nrm = norms(k);
168 }
169 }
170
171 // Swap current row with chosen pivot row
172 deep_swap(aa(j, _), aa(jpiv, _));
173 std::swap(norms(j), norms(jpiv));
174 std::swap(piv(j), piv(jpiv));
175
176 // Orthogonalize current rows (now the chosen pivot row) against all
177 // previously chosen rows
178 for (int k = 0; k < j; ++k) { aa(j, _) = aa(j, _) - aa(k, _) * nda::blas::dotc(aa(k, _), aa(j, _)); }
179
180 // Get norm of current row
181 nrm = normsq(aa(j, _));
182
183 // Terminate if sufficiently small, and return previously selected rows
184 // (not including current row)
185 if (nrm <= epssq) { return {aa(nda::range(0, j), _), norms(nda::range(0, j)), piv(nda::range(0, j))}; };
186
187 // Normalize current row
188 aa(j, _) /= sqrt(nrm);
189
190 // Orthogonalize remaining rows against current row
191 for (int k = j + 1; k < m; ++k) {
192 if (norms(k) <= epssq) { continue; } // Can skip rows with norm less than tolerance
193 aa(k, _) = aa(k, _) - aa(j, _) * nda::blas::dotc(aa(j, _), aa(k, _));
194 norms(k) = normsq(aa(k, _));
195 }
196 }
197
198 return {aa(nda::range(maxrnk), _), norms(nda::range(maxrnk)), piv(nda::range(maxrnk))};
199 }

231

233
234 // Copy input data, re-ordering rows to make symmetric rows adjacent.
235 auto aa = typename T::regular_type(m, n);
236 aa(nda::range(0, m, 2), _) = a(nda::range(0, m / 2), _);
237 aa(nda::range(1, m, 2), _) = a(nda::range(m - 1, m / 2 - 1, -1), _);
238
239 // Compute norms of rows of input matrix, and rescale eps tolerance
240 auto norms = nda::vector<double>(m);
241 double epssq = eps * eps;
242 for (int j = 0; j < m; ++j) { norms(j) = normsq(aa(j, _)); }
243
244 // Begin pivoted double Gram-Schmidt procedure
245 int jpiv = 0;
246 double nrm = 0;
247 auto piv = nda::arange(0, m);
248 piv(nda::range(0, m, 2)) = nda::arange(0, m / 2); // Re-order pivots to match re-ordered input matrix
249 piv(nda::range(1, m, 2)) = nda::arange(m - 1, m / 2 - 1, -1);
250
251 if (maxrnk % 2 != 0) { // If n < m and n is odd, decrease maxrnk to maintain symmetry
252 maxrnk -= 1;
253 }
254
255 for (int j = 0; j < maxrnk; j += 2) {
256
257 // Find next pair of pivots
258 jpiv = j;
259 nrm = norms(j) + norms(j + 1);
260 for (int k = j + 2; k < m; k += 2) {
261 if (norms(k) + norms(k + 1) > nrm) {
262 jpiv = k;
263 nrm = norms(k) + norms(k + 1);
264 }
265 }
266
267 // Swap current row pair with chosen pivot row pair
268 deep_swap(aa(j, _), aa(jpiv, _));
269 deep_swap(aa(j + 1, _), aa(jpiv + 1, _));
270 std::swap(norms(j), norms(jpiv));
271 std::swap(norms(j + 1), norms(jpiv + 1));
272 std::swap(piv(j), piv(jpiv));
273 std::swap(piv(j + 1), piv(jpiv + 1));
274
275 // Orthogonalize current row (now the first chosen pivot row) against all
276 // previously chosen rows
277 for (int k = 0; k < j; ++k) { aa(j, _) = aa(j, _) - aa(k, _) * nda::blas::dotc(aa(k, _), aa(j, _)); }
278
279 // Get norm of current row
280 nrm = normsq(aa(j, _));
281
282 // Terminate if sufficiently small, and return previously selected rows
283 // (not including current row)
284 if (nrm <= epssq) { return {aa(nda::range(0, j), _), norms(nda::range(0, j)), piv(nda::range(0, j))}; };
285
286 // Normalize current row
287 aa(j, _) /= sqrt(nrm);
288
289 // Orthogonalize remaining rows against current row
290 for (int k = j + 1; k < m; ++k) {
291 if (norms(k) <= epssq) { continue; } // Can skip rows with norm less than tolerance
292 aa(k, _) = aa(k, _) - aa(j, _) * nda::blas::dotc(aa(j, _), aa(k, _));
293 norms(k) = normsq(aa(k, _));
294 }
295
296 // Orthogonalize current row (now the second chosen pivot row) against all
297 // previously chosen rows
298 for (int k = 0; k < j + 1; ++k) { aa(j + 1, _) = aa(j + 1, _) - aa(k, _) * nda::blas::dotc(aa(k, _), aa(j + 1, _)); }
299
300 // Normalize current row
301 aa(j + 1, _) /= sqrt(normsq(aa(j + 1, _)));
302
303 // Orthogonalize remaining rows against current row
304 for (int k = j + 2; k < m; ++k) {
305 if (norms(k) <= epssq) { continue; } // Can skip rows with norm less than tolerance
306 aa(k, _) = aa(k, _) - aa(j + 1, _) * nda::blas::dotc(aa(j + 1, _), aa(k, _));
307 norms(k) = normsq(aa(k, _));
308 }
309 }
310
311 return {aa(nda::range(maxrnk), _), norms(nda::range(maxrnk)), piv(nda::range(maxrnk))};

341 std::tuple<typename T::regular_type, nda::vector<double>, nda::vector<int>> pivrgs_sym(T const &a, int r) {
342
343 // Get matrix dimensions
344 auto [m, n] = a.shape();
345
346 if (m % 2 == 1 && r % 2 == 0) { throw std::runtime_error("If input matrix has odd number of rows, r must be odd."); }
347 if (r % 2 == 1 && m % 2 == 0) { throw std::runtime_error("If r is odd, input matrix must have odd number of rows."); }
348 if (r > m || r > n + 1) { throw std::runtime_error("r must be less than or equal to min(m,n+1)."); }
349 if (r == n + 1 && (n % 2 == 1 || n > m)) { throw std::runtime_error("If r = n+1, n must be even and less than or equal to m."); }
350
351 // Copy input data, re-ordering rows to make symmetric rows adjacent. If m
352 // odd, put middle row at the end.
353 auto aa = typename T::regular_type(m, n);
354 if (m % 2 == 0) {
355 aa(nda::range(0, m, 2), _) = a(nda::range(0, m / 2), _);
356 aa(nda::range(1, m, 2), _) = a(nda::range(m - 1, m / 2 - 1, -1), _);
357 } else {
358 aa(0, _) = a((m - 1) / 2, _);
359 aa(nda::range(1, m, 2), _) = a(nda::range(0, (m - 1) / 2), _);
360 aa(nda::range(2, m, 2), _) = a(nda::range(m - 1, (m - 1) / 2, -1), _);
361 //aa(m - 1, _) = a((m - 1) / 2, _);
362 }
363
364 // Compute norms of rows of input matrix
365 auto norms = nda::vector<double>(m);
366 for (int j = 0; j < m; ++j) { norms(j) = normsq(aa(j, _)); }
367
368 // Begin pivoted double Gram-Schmidt procedure
369 int jpiv = 0;
370 double nrm = 0;
371 auto piv = nda::arange(0, m);
372 if (m % 2 == 0) {
373 piv(nda::range(0, m, 2)) = nda::arange(0, m / 2); // Re-order pivots to match re-ordered input matrix
374 piv(nda::range(1, m, 2)) = nda::arange(m - 1, m / 2 - 1, -1);
375 } else {
376 piv(0) = (m - 1) / 2;
377 piv(nda::range(1, m, 2)) = nda::arange(0, (m - 1) / 2);
378 piv(nda::range(2, m, 2)) = nda::arange(m - 1, (m - 1) / 2, -1);
379 //piv(m - 1) = (m - 1) / 2;
380 }
381
382 // If m odd, first choose middle row (now last row) as first pivot
383
384 if (m % 2 == 1) {
385 // Normalize
386 aa(0, _) /= sqrt(normsq(aa(0, _)));
387
388 // Orthogonalize remaining rows against current row
389 for (int k = 1; k < m; ++k) {
390 aa(k, _) = aa(k, _) - aa(0, _) * nda::blas::dotc(aa(0, _), aa(k, _));
391 norms(k) = normsq(aa(k, _));
392 }
393 }
394
395 int jstrt = (m % 2 == 1) ? 1 : 0;
396 // Then proceed with pivoted GS algorithm as normal
397 for (int j = jstrt; j < r; j += 2) {
398
399 // Find next pair of pivots
400 jpiv = j;
401 nrm = norms(j) + norms(j + 1);
402 for (int k = j + 2; k < m; k += 2) {
403 if (norms(k) + norms(k + 1) > nrm) {
404 jpiv = k;
405 nrm = norms(k) + norms(k + 1);
406 }
407 }
408
409 // Swap current row pair with chosen pivot row pair
410 deep_swap(aa(j, _), aa(jpiv, _));
411 deep_swap(aa(j + 1, _), aa(jpiv + 1, _));
412 std::swap(norms(j), norms(jpiv));
413 std::swap(norms(j + 1), norms(jpiv + 1));
414 std::swap(piv(j), piv(jpiv));
415 std::swap(piv(j + 1), piv(jpiv + 1));
416
417 // Orthogonalize current row (now the first chosen pivot row) against all
418 // previously chosen rows
419 for (int k = 0; k < j; ++k) { aa(j, _) = aa(j, _) - aa(k, _) * nda::blas::dotc(aa(k, _), aa(j, _)); }
420
421 // Normalize current row
422 aa(j, _) /= sqrt(normsq(aa(j, _)));
423
424 // Orthogonalize remaining rows against current row
425 for (int k = j + 1; k < m; ++k) {
426 aa(k, _) = aa(k, _) - aa(j, _) * nda::blas::dotc(aa(j, _), aa(k, _));
427 norms(k) = normsq(aa(k, _));
428 }
429
430 // Orthogonalize current row (now the second chosen pivot row) against all
431 // previously chosen rows
432 for (int k = 0; k < j + 1; ++k) { aa(j + 1, _) = aa(j + 1, _) - aa(k, _) * nda::blas::dotc(aa(k, _), aa(j + 1, _)); }
433
434 // Normalize current row
435 aa(j + 1, _) /= sqrt(normsq(aa(j + 1, _)));
436
437 // Orthogonalize remaining rows against current row
438 for (int k = j + 2; k < m; ++k) {
439 aa(k, _) = aa(k, _) - aa(j + 1, _) * nda::blas::dotc(aa(j + 1, _), aa(k, _));
440 norms(k) = normsq(aa(k, _));
441 }
442 }
443
444 return {aa(nda::range(r), _), norms(nda::range(r)), piv(nda::range(r))};
445 }

534 template <nda::MemoryArray Ta, nda::MemoryArray Tb, nda::Scalar Sa = nda::get_value_t<Ta>, nda::Scalar Sb = nda::get_value_t<Tb>,
535 nda::Scalar S = std::common_type_t<Sa, Sb>>
536 nda::array<S, Ta::rank + Tb::rank - 2> arraymult(Ta const &a, Tb const &b) {
537
538 // Get ranks of input arrays
539 constexpr int ra = Ta::rank;
540 constexpr int rb = Tb::rank;
541
542 // Get inner dimensions of input arrays
543 int p = a.shape(ra - 1);
544 if (b.shape(0) != p) throw std::runtime_error("last dim of a != first dim of b");
545
546 // Get product of outer dimensions of input arrays
547 int m = a.size() / p;
548 int n = b.size() / p;
549
550 // Reshape input arrays to 2D arrays
551 auto a_reshaped = nda::reshape(a, m, p);
552 auto b_reshaped = nda::reshape(b, p, n);
553
554 // Get shape of output array
555 auto c_shape = std::array<int, ra + rb - 2>();
556 for (int i = 0; i < ra - 1; ++i) { c_shape[i] = a.shape(i); }
557 for (int i = ra - 1; i < ra + rb - 2; ++i) { c_shape[i] = b.shape(i - ra + 2); }
558
559 // Compute the contraction, reshape, and return
560 return reshape(matmul(a_reshaped, b_reshaped), c_shape);
561 }

586 S adapgl(std::function<nda::array<S, 1>(nda::array<double, 1>)> f, double a, double b, double tol, nda::vector<double> xgl,
587 nda::vector<double> wgl) {
588
589 int maxnde = 100000; // Max nodes in integration tree
590 int maxstk = 1000; // Max stack size
591
592 S sum = 0; // Integral
593
594 // Initialize stack: there is a stack of interval endpoints associated w/
595 // each node in the integration tree, and a corresponding stack of values
596 // associated w/ each such interval
597
598 auto endpt = nda::array<double, 2>(maxstk, 2); // Stack of interval endpoints
599 auto sums = nda::array<S, 1>(maxstk); // Stack of sums associated w/ each interval
600
601 int istk = 0; // Stack pointer
602 int nnde = 1; // Number of nodes in integration tree
603
604 endpt(istk, 0) = a; // Root node interval is [a,b]
605 endpt(istk, 1) = b;
606
607 double h = (b - a) / 2; // Interval half-width
608 sums(istk) = h * nda::blas::dotc(wgl, f(xgl * h + (a + b) / 2)); // Integral of f on root node interval
609
610 double aa = 0, bb = 0, cc = 0; // Interval endpoints, midpoint
611 S sum1 = 0, sum2 = 0; // Integrals on subintervals
612
613 // Main loop
614 while (true) {
615
616 aa = endpt(istk, 0); // Get interval endpoints
617 bb = endpt(istk, 1);
618 cc = (aa + bb) / 2; // Get midpoint
619
620 // Get integral on each subinterval
621 h = (cc - aa) / 2;
622 sum1 = h * nda::blas::dotc(wgl, f(xgl * h + (aa + cc) / 2));
623 sum2 = h * nda::blas::dotc(wgl, f(xgl * h + (cc + bb) / 2));
624
625 // If error on subintervals is less than tolerance, add to integral and
626 // move to next node
627
628 if (std::abs(sum1 + sum2 - sums(istk)) < tol) {
629 sum += sum1 + sum2; // Add to subintervals to integral
630 istk--; // Remove current interval from stack
631 if (istk < 0) { return sum; } // If stack is empty, return integral
632 } else { // Integral on current interval not converged
633
634 // Remove current interval from stack, and add subintervals
635 endpt(istk, 1) = cc;
636 endpt(istk + 1, 0) = cc;
637 endpt(istk + 1, 1) = bb;
638 sums(istk) = sum1;
639 sums(istk + 1) = sum2;
640
641 istk++;
642 nnde += 2;
643
644 if (istk > maxstk) { throw std::runtime_error("integration stack too large"); }
645 if (nnde > maxnde) { throw std::runtime_error("integration tree too large"); }
646 }
647 }
