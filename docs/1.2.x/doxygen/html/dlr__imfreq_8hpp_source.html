82  template <nda::MemoryArray T> typename T::regular_type vals2coefs(T const &g) const {
83 
84  // MemoryArray type can be nda vector, matrix, array, or view of any of
85  // these; taking a regular_type converts, for example, a matrix view to a
86  // matrix.
87 
88  if (niom != g.shape(0)) throw std::runtime_error("First dim of g != # DLR imaginary frequency nodes.");
89 
90  // Make a copy of the data in Fortran Layout as required by getrs
91  auto gf = nda::array<get_value_t<T>, get_rank<T>, F_layout>(g);
92 
93  // Reshape as matrix_view with r rows
94  auto gfv = nda::reshape(gf, niom, g.size() / niom);
95 
96  // Solve linear system (multiple right hand sides) to convert vals ->
97  // coeffs
98  if (niom == r) {
99  nda::lapack::getrs(if2cf.lu, gfv, if2cf.piv);
100  } else { // Non-square system---use least squares solver
101  auto s = nda::vector<double>(r); // Not needed
102  double rcond = 0; // Not needed
103  int rank = 0; // Not needed
104  nda::lapack::gelss(nda::matrix<dcomplex, F_layout>(cf2if), gfv, s, rcond, rank);
105  }
106 
107  return gf(nda::range(r), nda::ellipsis());
108  }

123  template <nda::MemoryArray T, nda::Scalar S = nda::get_value_t<T>> make_cplx_t<T> coefs2vals(T const &gc) const {
124 
125  if (r != gc.shape(0)) throw std::runtime_error("First dim of gc != DLR rank r.");
126 
127  // Reshape gc to a matrix w/ first dimension r
128  auto gc_rs = nda::reshape(gc, r, gc.size() / r);
129 
130  // Apply coeffs -> vals matrix
131  auto g = cf2if * nda::matrix_const_view<S>(gc_rs);
132 
133  // Get output shape
134  std::array<long, T::rank> shape_out;
135  shape_out[0] = niom;
136  for (int i = 1; i < T::rank; ++i) { shape_out[i] = gc.shape(i); }
137 
138  // Reshape to original dimensions and return
139  return nda::reshape(g, shape_out);
140  }

156  template <nda::MemoryArray T, nda::Scalar S = nda::get_value_t<T>> auto coefs2eval(T const &gc, int n) const {
157 
158  if (r != gc.shape(0)) throw std::runtime_error("First dim of g != DLR rank r.");
159 
160  // Scalar-valued Green's functions are handled differently than matrix-valued Green's functions
161 
162  if constexpr (T::rank == 1) {
163 
164  // Evaluate DLR expansion
165  std::complex<double> g = 0;
166  for (int l = 0; l < r; ++l) { g += k_if(n, dlr_rf(l), statistic) * gc(l); }
167 
168  return g;
169  } else {
170 
171  // Reshape gc to matrix w/ first dimension r
172  auto gc_rs = nda::reshape(gc, r, gc.size() / r);
173 
174  // Get output shape
175  std::array<long, T::rank - 1> shape_out;
176  for (int i = 0; i < T::rank - 1; ++i) { shape_out[i] = gc.shape(i + 1); }
177 
178  // Get vector of evaluation of DLR expansion at a point
179  auto kvec = build_evalvec(n);
180 
181  // Evaluate DLR expansion, reshape to original dimensions (with first
182  // dimension summed out), and return
183  return nda::reshape(transpose(nda::matrix_const_view<S>(gc_rs)) * kvec, shape_out);
184  }
185  }
