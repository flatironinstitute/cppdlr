159 template <nda::MemoryArray T, nda::Scalar S = nda::get_value_t<T>> auto coefs2eval(T const &gc, double t) const {
160
161 if (r != gc.shape(0)) throw std::runtime_error("First dim of g != DLR rank r.");
162
163 // Scalar-valued Green's functions are handled differently than matrix-valued Green's functions
164
165 if constexpr (T::rank == 1) {
166
167 // TODO: can be further optimized to reduce # exponential evals.
168 // Evaluate DLR expansion
169 S g = 0;
170 if (t >= 0) {
171 for (int l = 0; l < r; ++l) { g += k_it_abs(t, dlr_rf(l)) * gc(l); }
172 } else {
173 for (int l = 0; l < r; ++l) { g += k_it_abs(-t, -dlr_rf(l)) * gc(l); }
174 }
175
176 return g;
177 } else {
178
179 // Reshape gc to matrix w/ first dimension r
180 auto gc_rs = nda::reshape(gc, r, gc.size() / r);
181
182 // Get output shape
183 std::array<long, T::rank - 1> shape_out;
184 for (int i = 0; i < T::rank - 1; ++i) { shape_out[i] = gc.shape(i + 1); }
185
186 // Get vector of evaluation of DLR expansion at a point
187 auto kvec = build_evalvec(t);
188
189 // Evaluate DLR expansion, reshape to original dimensions (with first
190 // dimension summed out), and return
191 return nda::reshape(transpose(nda::matrix_const_view<S>(gc_rs)) * kvec, shape_out);
192 }

218 template <nda::MemoryArray T, nda::Scalar S = nda::get_value_t<T>>
219 typename T::regular_type fitvals2coefs(nda::vector_const_view<double> t, T const &g) const {
220
221 // Check first dimension of g equal to length of t
222 int n = t.size();
223 if (n != g.shape(0)) throw std::runtime_error("First dim of g must be equal to length of t.");
224
225 // Get matrix for least squares fitting: columns are DLR basis functions
226 // evaluating at data points t. Must built in Fortran layout for
227 // compatibility with LAPACK.
228 auto kmat = nda::matrix<S, nda::F_layout>(n, r); // Make sure matrix has same scalar type as g
229 for (int j = 0; j < r; ++j) {
230 for (int i = 0; i < n; ++i) { kmat(i, j) = k_it(t(i), dlr_rf(j)); }
231 }
232
233 // Reshape g to matrix w/ first dimension n, and put in Fortran layout for
234 // compatibility w/ LAPACK
235 auto g_rs = nda::matrix<S, nda::F_layout>(nda::reshape(g, n, g.size() / n));
236
237 // Solve least squares problem to obtain DLR coefficients
238 auto s = nda::vector<double>(r); // Singular values (not needed)
239 int rank = 0; // Rank of system matrix (not needed)
240 nda::lapack::gelss(kmat, g_rs, s, 0.0, rank);
241
242 auto gc_shape = g.shape(); // Output shape is same as g...
243 gc_shape[0] = r; // ...with first dimension n replaced by r
244 auto gc = typename T::regular_type(gc_shape); // Output array: output type is same as g
245 reshape(gc, r, g.size() / n) = g_rs(nda::range(r), _); // Place result in output array
246
247 return gc;
248 }

262

264
265 // Initialize reflection matrix, if it hasn't been done already
266 if (refl.empty()) { reflect_init(); }
267

269 return matmul(refl, g);

306 template <nda::MemoryArray T, nda::Scalar S = nda::get_value_t<T>>
307 typename T::regular_type convolve(double beta, statistic_t statistic, T const &fc, T const &gc, bool time_order = false) const {
308
309 if (r != fc.shape(0) || r != gc.shape(0)) throw std::runtime_error("First dim of input arrays must be equal to DLR rank r.");
310 if (fc.shape() != gc.shape()) throw std::runtime_error("Input arrays must have the same shape.");
311
312 // TODO: implement bosonic case and remove
313 if (statistic == 0) throw std::runtime_error("imtime_ops::convolve not yet implemented for bosonic Green's functions.");
314
315 // Initialize convolution, if it hasn't been done already
316 if (!time_order & hilb.empty()) { convolve_init(); }
317 if (time_order & thilb.empty()) { tconvolve_init(); }
318
319 // Get view of helper matrices based on time_order flag
320 auto hilb_v = (time_order ? nda::matrix_const_view<double>(thilb) : nda::matrix_const_view<double>(hilb));
321 auto tcf2it_v = (time_order ? nda::matrix_const_view<double>(ttcf2it) : nda::matrix_const_view<double>(tcf2it));
322
323 if constexpr (T::rank == 1) { // Scalar-valued Green's function
324
325 // Take array view of fc and gc
326 auto fca = nda::array_const_view<S, 1>(fc);
327 auto gca = nda::array_const_view<S, 1>(gc);
328
329 // Diagonal contribution
330 auto h = matvecmul(tcf2it_v, make_regular(fca * gca));
331
332 // Off-diagonal contribution
333 auto tmp = fca * arraymult(hilb_v, gca) + gca * arraymult(hilb_v, fca);
334 return beta * (h + matvecmul(cf2it, make_regular(tmp)));
335
336 } else if (T::rank == 3) { // Matrix-valued Green's function
337
338 // Diagonal contribution
339 auto fcgc = nda::array<S, 3>(fc.shape()); // Product of coefficients of f and g
340 for (int i = 0; i < r; ++i) { fcgc(i, _, _) = matmul(fc(i, _, _), gc(i, _, _)); }
341 auto h = arraymult(tcf2it_v, fcgc);
342
343 // Off-diagonal contribution
344 auto tmp1 = arraymult(hilb_v, fc);
345 auto tmp2 = arraymult(hilb_v, gc);
346 for (int i = 0; i < r; ++i) { tmp1(i, _, _) = matmul(tmp1(i, _, _), gc(i, _, _)) + matmul(fc(i, _, _), tmp2(i, _, _)); }
347
348 return beta * (h + arraymult(cf2it, tmp1));
349
350 } else {
351 throw std::runtime_error("Input arrays must be rank 1 (scalar-valued Green's function) or 3 (matrix-valued Green's function).");
352 }
353 }

376 template <nda::MemoryMatrix Tf, nda::MemoryArray Tg, nda::Scalar Sf = nda::get_value_t<Tf>, nda::Scalar Sg = nda::get_value_t<Tg>,
377 nda::Scalar S = typename std::common_type<Sf, Sg>::type>
378 typename Tg::regular_type convolve(Tf const &fconv, Tg const &g) const {
379
380 if (r != g.shape(0)) throw std::runtime_error("First dim of input g must be equal to DLR rank r.");
381
382 if constexpr (Tg::rank == 1) { // Scalar-valued Green's function
383
384 if (fconv.shape(1) != g.shape(0)) throw std::runtime_error("Input array dimensions incompatible.");
385
386 return matvecmul(fconv, g);
387
388 } else if (Tg::rank == 3) { // Matrix-valued Green's function
389
390 if (fconv.shape(1) != g.shape(0) * g.shape(1)) throw std::runtime_error("Input array dimensions incompatible.");
391
392 int norb1 = g.shape(1);
393 int norb2 = g.shape(2);
394
395 auto h = nda::array<S, 3>(r, norb1, norb2);
396 reshape(h, r * norb1, norb2) = matmul(fconv, reshape(g, r * norb1, norb2));
397
398 return h;
399
400 } else {
401 throw std::runtime_error("Input arrays must be rank 1 (scalar-valued Green's function) or 3 (matrix-valued Green's function).");
402 }
403 }

449 template <nda::MemoryArray T, nda::Scalar S = nda::get_value_t<T>>
450 nda::matrix<S> convmat(double beta, statistic_t statistic, T const &fc, bool time_order = false) const {
451
452 int n, m;
453
454 if constexpr (T::rank == 1) { // Scalar-valued Green's function
455 n = r;
456 m = r;
457 } else if (T::rank == 3) { // Matrix-valued Green's function
458 n = r * fc.shape(1);
459 m = r * fc.shape(2);
460 } else {
461 throw std::runtime_error("Input arrays must be rank 1 (scalar-valued Green's function) or 3 (matrix-valued Green's function).");
462 }
463
464 auto fconv = nda::matrix<S, nda::C_layout>(n, m); // Matrix of convolution by f
465 convmat_inplace(nda::matrix_view<S, nda::C_layout>(fconv), beta, statistic, fc, time_order);
466
467 return fconv;
468 }

519 template <nda::MemoryArray T, nda::Scalar S = nda::get_value_t<T>>
520 void convmat_inplace(nda::matrix_view<S, nda::C_layout> fconv, double beta, statistic_t statistic, T const &fc, bool time_order = false) const {
521
522 if (r != fc.shape(0)) throw std::runtime_error("First dim of input array must be equal to DLR rank r.");
523
524 // TODO: implement bosonic case and remove
525 if (statistic == 0) throw std::runtime_error("imtime_ops::convmat not yet implemented for bosonic Green's functions.");
526
527 // Initialize convolution, if it hasn't been done already
528 if (!time_order & hilb.empty()) { convolve_init(); }
529 if (time_order & thilb.empty()) { tconvolve_init(); }
530
531 // Get view of helper matrices based on time_order flag
532 auto hilb_v = (time_order ? nda::matrix_const_view<double>(thilb) : nda::matrix_const_view<double>(hilb));
533 auto tcf2it_v = (time_order ? nda::matrix_const_view<double>(ttcf2it) : nda::matrix_const_view<double>(tcf2it));
534
535 if constexpr (T::rank == 1) { // Scalar-valued Green's function
536
537 if (fconv.shape(0) != r || fconv.shape(1) != r) throw std::runtime_error("Matrix shape must be equal to DLR rank (r,r).");
538
539 // Diagonal contribution (given by diag(tau_k) * K(tau_k, om_l) * diag(fc_l))
540 for (int k = 0; k < r; ++k) {
541 for (int l = 0; l < r; ++l) { fconv(k, l) = tcf2it_v(k, l) * fc(l); }
542 }
543
544 // Off-diagonal contribution (given by K * (diag(hilb*fc) +
545 // (diag(fc)*hilb)), where K is the matrix K(dlr_it(k), dlr_rf(l))
546 auto tmp1 = matvecmul(hilb_v, fc); // hilb * fc
547 auto tmp2 = nda::matrix<S>(r, r);
548 for (int k = 0; k < r; ++k) {
549 for (int l = 0; l < r; ++l) {
550 tmp2(k, l) = fc(k) * hilb_v(k, l); // diag(fc) * hilb
551 }
552 tmp2(k, k) += tmp1(k); // diag(fc)*hilb + diag(hilb*fc)
553 }
554 fconv += matmul(cf2it, tmp2);
555
556 // Then precompose with DLR grid values to DLR coefficients matrix
557 if constexpr (nda::is_complex_v<S>) { // getrs requires matrix and rhs to have same value type
558 nda::lapack::getrs(transpose(it2cf.zlu), fconv, it2cf.piv); // Note: lapack effectively tranposes fconv by fortran reordering here
559 } else {
560 // getrs requires matrix and rhs to have same value type
561 nda::lapack::getrs(transpose(it2cf.lu), fconv, it2cf.piv);
562 }
563
564 fconv *= beta;
565
566 } else if (T::rank == 3) { // Matrix-valued Green's function
567
568 int norb1 = fc.shape(1);
569 int norb2 = fc.shape(2);
570
571 if (fconv.shape(0) != r * norb1 || fconv.shape(1) != r * norb2)
572 throw std::runtime_error("Matrix shape must be equal to DLR rank times norbs (r*norb1,r*norb2).");
573
574 auto fconv_rs = nda::reshape(fconv, r, norb1, r, norb2); // Array view to index into fconv for conevenience
575
576 // Diagonal contribution (given by diag(tau_k) * K(tau_k, om_l) * diag(fc_l))
577 for (int k = 0; k < r; ++k) {
578 for (int l = 0; l < r; ++l) { fconv_rs(k, _, l, _) = tcf2it_v(k, l) * fc(l, _, _); }
579 }
580
581 // Off-diagonal contribution (given by K * (diag(hilb*fc) +
582 // (diag(fc)*hilb)), where K is the matrix K(dlr_it(k), dlr_rf(l))
583 auto tmp1 = arraymult(hilb_v, fc); // hilb * fc
584 auto tmp2 = nda::array<S, 4>(r, norb1, r, norb2);
585 for (int k = 0; k < r; ++k) {
586 for (int l = 0; l < r; ++l) {
587 tmp2(k, _, l, _) = fc(k, _, _) * hilb_v(k, l); // diag(fc) * hilb
588 }
589 tmp2(k, _, k, _) += tmp1(k, _, _); // diag(fc)*hilb + diag(hilb*fc)
590 }
591 fconv_rs += arraymult(cf2it, tmp2);
592
593 // Then precompose with DLR grid values to DLR coefficients matrix
594
595 // Transpose last two indices to put make column DLR index the last
596 // index
597 auto fconvtmp = nda::matrix<S>(r * norb1 * norb2, r);
598 auto fconvtmp_rs = nda::reshape(fconvtmp, r, norb1, norb2, r);
599 for (int i = 0; i < norb2; ++i) {
600 for (int k = 0; k < r; ++k) { fconvtmp_rs(_, _, i, k) = fconv_rs(_, _, k, i); }
601 }
602
603 // Do the solve
604 if constexpr (nda::is_complex_v<S>) { // getrs requires matrix and rhs to have same value type
605 nda::lapack::getrs(transpose(it2cf.zlu), fconvtmp, it2cf.piv); // Note: lapack effectively tranposes fconv by fortran reordering here
606 } else {
607 // getrs requires matrix and rhs to have same value type
608 nda::lapack::getrs(transpose(it2cf.lu), fconvtmp, it2cf.piv);
609 }
610
611 // Transpose back
612 for (int i = 0; i < norb2; ++i) {
613 for (int k = 0; k < r; ++k) { fconv_rs(_, _, k, i) = fconvtmp_rs(_, _, i, k); }
614 }
615
616 fconv *= beta;
617
618 } else {
619 throw std::runtime_error("Input arrays must be rank 1 (scalar-valued Green's function) or 3 (matrix-valued Green's function).");
620 }
621 }

644 template <nda::MemoryArray T, nda::Scalar S = nda::get_value_t<T>> S innerprod(T const &fc, T const &gc) const {
645
646 if (r != fc.shape(0) || r != gc.shape(0)) throw std::runtime_error("First dim of input arrays must be equal to DLR rank r.");
647 if (fc.shape() != gc.shape()) throw std::runtime_error("Input arrays must have the same shape.");
648
649 // Initialize inner product matrix, if it hasn't been done already
650 if (ipmat.empty()) { innerprod_init(); }
651
652 S ip = 0;
653 if constexpr (T::rank == 1) { // Scalar-valued Green's function
654 ip = nda::blas::dotc(fc, matvecmul(ipmat, gc));
655 } else if (T::rank == 3) { // Matrix-valued Green's function
656 int norb = fc.shape(1);
657 // Compute inner product
658 for (int i = 0; i < norb; ++i) {
659 for (int j = 0; j < norb; ++j) { ip += nda::blas::dotc(fc(_, i, j), matvecmul(ipmat, gc(_, i, j))); }
660 }
661 } else {
662 throw std::runtime_error("Input arrays must be rank 1 (scalar-valued Green's function) or 3 (matrix-valued Green's function).");
663 }
664
665 return ip;
666 }

745 void convolve_init() const {
746
747 hilb = nda::matrix<double>(r, r);
748 tcf2it = nda::matrix<double>(r, r);
749
750 // "Discrete Hilbert transform" matrix -(1-delta_jk)/(dlr_rf(j) -
751 // dlr_rf(k)), scaled by beta
752 for (int j = 0; j < r; ++j) {
753 for (int k = 0; k < r; ++k) {
754 if (j == k) {
755 hilb(j, k) = 0;
756 } else {
757 hilb(j, k) = 1.0 / (dlr_rf(j) - dlr_rf(k));
758 }
759 }
760 }
761
762 // Matrix which applies DLR coefficients to imaginary time grid values
763 // transformation matrix, and then multiplies the result by tau, the
764 // imaginary time variable
765 for (int j = 0; j < r; ++j) {
766 for (int k = 0; k < r; ++k) {
767 if (dlr_it(j) > 0) {
768 tcf2it(j, k) = -(dlr_it(j) + k_it(1.0, dlr_rf(k))) * cf2it(j, k);
769 } else {
770 tcf2it(j, k) = -(dlr_it(j) - k_it(0.0, dlr_rf(k))) * cf2it(j, k);
771 }
772 }
773 }
774 }

785 void tconvolve_init() const {
786
787 thilb = nda::matrix<double>(r, r);
788 ttcf2it = nda::matrix<double>(r, r);
789
790 // "Discrete Hilbert transform" matrix
791 // -(1-delta_jk)*K(0,dlr_rf(k))/(dlr_rf(j) - dlr_rf(k)) for time-ordered
792 // convolution, scaled by beta
793 for (int j = 0; j < r; ++j) {
794 for (int k = 0; k < r; ++k) {
795 if (j == k) {
796 thilb(j, k) = 0;
797 } else {
798 thilb(j, k) = k_it(0.0, dlr_rf(k)) / (dlr_rf(k) - dlr_rf(j));
799 }
800 }
801 }
802
803 // Matrix which applies K(0,dlr_rf(j)) multiplication, then DLR
804 // coefficients to imaginary time grid values transformation matrix, and
805 // then multiplies the result by tau, the imaginary time variable
806 for (int j = 0; j < r; ++j) {
807 for (int k = 0; k < r; ++k) {
808 if (dlr_it(j) > 0) {
809 ttcf2it(j, k) = dlr_it(j) * cf2it(j, k) * k_it(0.0, dlr_rf(k));
810 } else {
811 ttcf2it(j, k) = (1 + dlr_it(j)) * cf2it(j, k) * k_it(0.0, dlr_rf(k));
812 }
813 }
814 }

824 void innerprod_init() const {
825
826 ipmat = nda::matrix<double>(r, r);
827
828 // Matrix of inner product of two DLR expansions
829 double ssum = 0;
830 for (int k = 0; k < r; ++k) {
831 for (int l = 0; l < r; ++l) {
832 ssum = dlr_rf(k) + dlr_rf(l);
833 if (ssum == 0) {
834 ipmat(k, l) = k_it(0.0, dlr_rf(k)) * k_it(0.0, dlr_rf(l));
835 } else if (std::abs(ssum) < 1) {
836 ipmat(k, l) = -k_it(0.0, dlr_rf(k)) * k_it(0.0, dlr_rf(l)) * std::expm1(-ssum) / ssum;
837 } else {
838 ipmat(k, l) = (k_it(0.0, dlr_rf(k)) * k_it(0.0, dlr_rf(l)) - k_it(1.0, dlr_rf(k)) * k_it(1.0, dlr_rf(l))) / ssum;
839 }
840 }

853 void reflect_init() const {
854
855 refl = nda::matrix<double>(r, r);
856
857 // Matrix of reflection acting on DLR coefficients and returning values at
858 // DLR nodes
859 for (int i = 0; i < r; ++i) {
860 for (int j = 0; j < r; ++j) { refl(i, j) = k_it(-dlr_it(i), dlr_rf(j)); }
861 }
862
863 // Precompose with DLR values to coefficients matrix
864 nda::lapack::getrs(transpose(it2cf.lu), refl, it2cf.piv); // Lapack effectively transposes refl by fortran reordering here
865 }
