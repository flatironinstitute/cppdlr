105 std::tuple<typename T::regular_type, nda::vector<double>, nda::vector<int>> pivrgs(T const &a, double eps) {
106
107 auto _ = nda::range::all;
108
109 // Copy input data
110 auto aa = make_regular(a);
111
112 // Get matrix dimensions
113 auto [m, n] = aa.shape();
114 int maxrnk = std::min(m, n);
115
116 // Compute norms of rows of input matrix, and rescale eps tolerance
117 auto norms = nda::vector<double>(m);
118 double epssq = eps * eps;
119 for (int j = 0; j < m; ++j) { norms(j) = nda::real(nda::blas::dotc(aa(j, _), aa(j, _))); }
120
121 // Begin pivoted double Gram-Schmidt procedure
122 int jpiv = 0, jj = 0;
123 double nrm = 0;
124 auto piv = nda::arange(m);
125 auto tmp = nda::vector<S>(n);
126
127 for (int j = 0; j < maxrnk; ++j) {
128
129 // Find next pivot
130 jpiv = j;
131 nrm = norms(j);
132 for (int k = j + 1; k < m; ++k) {
133 if (norms(k) > nrm) {
134 jpiv = k;
135 nrm = norms(k);
136 }
137 }
138
139 // Swap current row with chosen pivot row
140 tmp = aa(j, _);
141 aa(j, _) = aa(jpiv, _);
142 aa(jpiv, _) = tmp;
143
144 nrm = norms(j);
145 norms(j) = norms(jpiv);
146 norms(jpiv) = nrm;
147
148 jj = piv(j);
149 piv(j) = piv(jpiv);
150 piv(jpiv) = jj;
151
152 // Orthogonalize current rows (now the chosen pivot row) against all
153 // previously chosen rows
154 for (int k = 0; k < j; ++k) { aa(j, _) = aa(j, _) - aa(k, _) * nda::blas::dotc(aa(k, _), aa(j, _)); }
155
156 // Get norm of current row
157 nrm = nda::real(nda::blas::dotc(aa(j, _), aa(j, _)));
158 //nrm = nda::norm(aa(j, _));
159
160 // Terminate if sufficiently small, and return previously selected rows
161 // (not including current row)
162 if (nrm <= epssq) { return {aa(nda::range(0, j), _), norms(nda::range(0, j)), piv(nda::range(0, j))}; };
163
164 // Normalize current row
165 aa(j, _) = aa(j, _) * (1 / sqrt(nrm));
166
167 // Orthogonalize remaining rows against current row
168 for (int k = j + 1; k < m; ++k) {
169 if (norms(k) <= epssq) { continue; } // Can skip rows with norm less than tolerance
170 aa(k, _) = aa(k, _) - aa(j, _) * nda::blas::dotc(aa(j, _), aa(k, _));
171 norms(k) = nda::real(nda::blas::dotc(aa(k, _), aa(k, _)));
172 }
173 }
174
175 return {aa(nda::range(maxrnk), _), norms(nda::range(maxrnk)), piv(nda::range(maxrnk))};

553 template <nda::MemoryArray Ta, nda::MemoryArray Tb, nda::Scalar Sa = nda::get_value_t<Ta>, nda::Scalar Sb = nda::get_value_t<Tb>,
554 nda::Scalar S = typename std::common_type<Sa, Sb>::type>
555 nda::array<S, Ta::rank + Tb::rank - 2> arraymult(Ta const &a, Tb const &b) {
556
557 // Get ranks of input arrays
558 constexpr int ra = Ta::rank;
559 constexpr int rb = Tb::rank;
560
561 // Get inner dimensions of input arrays
562 int p = a.shape(ra - 1);
563 if (b.shape(0) != p) throw std::runtime_error("last dim of a != first dim of b");
564
565 // Get product of outer dimensions of input arrays
566 int m = a.size() / p;
567 int n = b.size() / p;
568
569 // Reshape input arrays to 2D arrays
570 auto a_reshaped = nda::reshape(a, m, p);
571 auto b_reshaped = nda::reshape(b, p, n);
572
573 // Get shape of output array
574 auto c_shape = std::array<int, ra + rb - 2>();
575 for (int i = 0; i < ra - 1; ++i) { c_shape[i] = a.shape(i); }
576 for (int i = ra - 1; i < ra + rb - 2; ++i) { c_shape[i] = b.shape(i - ra + 2); }
577
578 // Compute the contraction, reshape, and return
579 return reshape(matmul(a_reshaped, b_reshaped), c_shape);
580 }

603 template <typename S>
604 // TODO: require S is scalar
605 S adapgl(std::function<nda::array<S, 1>(nda::array<double, 1>)> f, double a, double b, double tol, nda::vector<double> xgl,
606 nda::vector<double> wgl) {
607
608 int maxnde = 100000; // Max nodes in integration tree
609 int maxstk = 1000; // Max stack size
610
611 S sum = 0; // Integral
612
613 // Initialize stack: there is a stack of interval endpoints associated w/
614 // each node in the integration tree, and a corresponding stack of values
615 // associated w/ each such interval
616
617 auto endpt = nda::array<double, 2>(maxstk, 2); // Stack of interval endpoints
618 auto sums = nda::array<S, 1>(maxstk); // Stack of sums associated w/ each interval
619
620 int istk = 0; // Stack pointer
621 int nnde = 1; // Number of nodes in integration tree
622
623 endpt(istk, 0) = a; // Root node interval is [a,b]
624 endpt(istk, 1) = b;
625
626 double h = (b - a) / 2; // Interval half-width
627 sums(istk) = h * nda::blas::dotc(wgl, f(xgl * h + (a + b) / 2)); // Integral of f on root node interval
628
629 double aa = 0, bb = 0, cc = 0; // Interval endpoints, midpoint
630 S sum1 = 0, sum2 = 0; // Integrals on subintervals
631
632 // Main loop
633 while (true) {
634
635 aa = endpt(istk, 0); // Get interval endpoints
636 bb = endpt(istk, 1);
637 cc = (aa + bb) / 2; // Get midpoint
638
639 // Get integral on each subinterval
640 h = (cc - aa) / 2;
641 sum1 = h * nda::blas::dotc(wgl, f(xgl * h + (aa + cc) / 2));
642 sum2 = h * nda::blas::dotc(wgl, f(xgl * h + (cc + bb) / 2));
643
644 // If error on subintervals is less than tolerance, add to integral and
645 // move to next node
646
647 if (std::abs(sum1 + sum2 - sums(istk)) < tol) {
648 sum += sum1 + sum2; // Add to subintervals to integral
649 istk--; // Remove current interval from stack
650 if (istk < 0) { return sum; } // If stack is empty, return integral
651 } else { // Integral on current interval not converged
652
653 // Remove current interval from stack, and add subintervals
654 endpt(istk, 1) = cc;
655 endpt(istk + 1, 0) = cc;
656 endpt(istk + 1, 1) = bb;
657 sums(istk) = sum1;
658 sums(istk + 1) = sum2;
659
660 istk++;
661 nnde += 2;
662
663 if (istk > maxstk) { throw std::runtime_error("integration stack too large"); }
664 if (nnde > maxnde) { throw std::runtime_error("integration tree too large"); }
665 }
666 }
667 }
