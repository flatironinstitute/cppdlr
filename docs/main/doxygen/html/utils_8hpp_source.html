112 template <nda::MemoryArrayOfRank<2> T, nda::Scalar S = nda::get_value_t<T>>
113 std::tuple<typename T::regular_type, nda::vector<double>, nda::vector<int>> pivrgs(T const &a, double eps) {
114
115 auto _ = nda::range::all;
116
117 // Copy input data
118 auto aa = make_regular(a);
119
120 // Get matrix dimensions
121 auto [m, n] = aa.shape();
122 int maxrnk = std::min(m, n);
123
124 // Compute norms of rows of input matrix, and rescale eps tolerance
125 auto norms = nda::vector<double>(m);
126 double epssq = eps * eps;
127 for (int j = 0; j < m; ++j) { norms(j) = normsq(aa(j, _)); }
128
129 // Begin pivoted double Gram-Schmidt procedure
130 int jpiv = 0;
131 double nrm = 0;
132 auto piv = nda::arange(m);
133 auto tmp = nda::vector<S>(n);
134
135 for (int j = 0; j < maxrnk; ++j) {
136
137 // Find next pivot
138 jpiv = j;
139 nrm = norms(j);
140 for (int k = j + 1; k < m; ++k) {
141 if (norms(k) > nrm) {
142 jpiv = k;
143 nrm = norms(k);
144 }
145 }
146
147 // Swap current row with chosen pivot row
148 deep_swap(aa(j, _), aa(jpiv, _));
149 std::swap(norms(j), norms(jpiv));
150 std::swap(piv(j), piv(jpiv));
151
152 // Orthogonalize current rows (now the chosen pivot row) against all
153 // previously chosen rows
154 for (int k = 0; k < j; ++k) { aa(j, _) = aa(j, _) - aa(k, _) * nda::blas::dotc(aa(k, _), aa(j, _)); }
155
156 // Get norm of current row
157 nrm = normsq(aa(j, _));
158
159 // Terminate if sufficiently small, and return previously selected rows
160 // (not including current row)
161 if (nrm <= epssq) { return {aa(nda::range(0, j), _), norms(nda::range(0, j)), piv(nda::range(0, j))}; };
162
163 // Normalize current row
164 aa(j, _) /= sqrt(nrm);
165
166 // Orthogonalize remaining rows against current row
167 for (int k = j + 1; k < m; ++k) {
168 if (norms(k) <= epssq) { continue; } // Can skip rows with norm less than tolerance
169 aa(k, _) = aa(k, _) - aa(j, _) * nda::blas::dotc(aa(j, _), aa(k, _));
170 norms(k) = normsq(aa(k, _));
171 }
172 }
173
174 return {aa(nda::range(maxrnk), _), norms(nda::range(maxrnk)), piv(nda::range(maxrnk))};
175 }

215 aa(nda::range(1, m, 2), _) = a(nda::range(m - 1, m / 2 - 1, -1), _);

220 for (int j = 0; j < m; ++j) { norms(j) = normsq(aa(j, _)); }
221

223 int jpiv = 0;
224 double nrm = 0;
225 auto piv = nda::arange(0, m);
226 piv(nda::range(0, m, 2)) = nda::arange(0, m / 2); // Re-order pivots to match re-ordered input matrix
227 piv(nda::range(1, m, 2)) = nda::arange(m - 1, m / 2 - 1, -1);
228
229 if (maxrnk % 2 != 0) { // If n < m and n is odd, decrease maxrnk to maintain symmetry
230 maxrnk -= 1;
231 }
232
233 for (int j = 0; j < maxrnk; j += 2) {
234
235 // Find next pair of pivots
236 jpiv = j;
237 nrm = norms(j) + norms(j + 1);
238 for (int k = j + 2; k < m; k += 2) {
239 if (norms(k) + norms(k + 1) > nrm) {
240 jpiv = k;
241 nrm = norms(k) + norms(k + 1);
242 }
243 }
244
245 // Swap current row pair with chosen pivot row pair
246 deep_swap(aa(j, _), aa(jpiv, _));
247 deep_swap(aa(j + 1, _), aa(jpiv + 1, _));
248 std::swap(norms(j), norms(jpiv));
249 std::swap(norms(j + 1), norms(jpiv + 1));
250 std::swap(piv(j), piv(jpiv));
251 std::swap(piv(j + 1), piv(jpiv + 1));
252
253 // Orthogonalize current row (now the first chosen pivot row) against all
254 // previously chosen rows
255 for (int k = 0; k < j; ++k) { aa(j, _) = aa(j, _) - aa(k, _) * nda::blas::dotc(aa(k, _), aa(j, _)); }
256
257 // Get norm of current row
258 nrm = normsq(aa(j, _));
259
260 // Terminate if sufficiently small, and return previously selected rows
261 // (not including current row)
262 if (nrm <= epssq) { return {aa(nda::range(0, j), _), norms(nda::range(0, j)), piv(nda::range(0, j))}; };
263
264 // Normalize current row
265 aa(j, _) /= sqrt(nrm);
266
267 // Orthogonalize remaining rows against current row
268 for (int k = j + 1; k < m; ++k) {
269 if (norms(k) <= epssq) { continue; } // Can skip rows with norm less than tolerance
270 aa(k, _) = aa(k, _) - aa(j, _) * nda::blas::dotc(aa(j, _), aa(k, _));
271 norms(k) = normsq(aa(k, _));
272 }
273
274 // Orthogonalize current row (now the second chosen pivot row) against all
275 // previously chosen rows
276 for (int k = 0; k < j + 1; ++k) { aa(j + 1, _) = aa(j + 1, _) - aa(k, _) * nda::blas::dotc(aa(k, _), aa(j + 1, _)); }
277
278 // Normalize current row
279 aa(j + 1, _) /= sqrt(normsq(aa(j + 1, _)));
280

282 for (int k = j + 2; k < m; ++k) {
283 if (norms(k) <= epssq) { continue; } // Can skip rows with norm less than tolerance

285 norms(k) = normsq(aa(k, _));
286 }
287 }
288

319 std::tuple<typename T::regular_type, nda::vector<double>, nda::vector<int>> pivrgs_sym(T const &a, int r) {
320
321 auto _ = nda::range::all;
322
323 // Get matrix dimensions
324 auto [m, n] = a.shape();
325
326 if (m % 2 == 1 && r % 2 == 0) { throw std::runtime_error("If input matrix has odd number of rows, r must be odd."); }
327 if (r % 2 == 1 && m % 2 == 0) { throw std::runtime_error("If r is odd, input matrix must have odd number of rows."); }
328 if (r > m || r > n + 1) { throw std::runtime_error("r must be less than or equal to min(m,n+1)."); }
329 if (r == n + 1 && (n % 2 == 1 || n > m)) { throw std::runtime_error("If r = n+1, n must be even and less than or equal to m."); }
330
331 // Copy input data, re-ordering rows to make symmetric rows adjacent. If m
332 // odd, put middle row at the end.
333 auto aa = typename T::regular_type(m, n);
334 if (m % 2 == 0) {
335 aa(nda::range(0, m, 2), _) = a(nda::range(0, m / 2), _);
336 aa(nda::range(1, m, 2), _) = a(nda::range(m - 1, m / 2 - 1, -1), _);
337 } else {
338 aa(0, _) = a((m - 1) / 2, _);
339 aa(nda::range(1, m, 2), _) = a(nda::range(0, (m - 1) / 2), _);
340 aa(nda::range(2, m, 2), _) = a(nda::range(m - 1, (m - 1) / 2, -1), _);
341 //aa(m - 1, _) = a((m - 1) / 2, _);
342 }
343
344 // Compute norms of rows of input matrix
345 auto norms = nda::vector<double>(m);
346 for (int j = 0; j < m; ++j) { norms(j) = normsq(aa(j, _)); }
347
348 // Begin pivoted double Gram-Schmidt procedure
349 int jpiv = 0;
350 double nrm = 0;
351 auto piv = nda::arange(0, m);
352 if (m % 2 == 0) {
353 piv(nda::range(0, m, 2)) = nda::arange(0, m / 2); // Re-order pivots to match re-ordered input matrix
354 piv(nda::range(1, m, 2)) = nda::arange(m - 1, m / 2 - 1, -1);
355 } else {
356 piv(0) = (m - 1) / 2;
357 piv(nda::range(1, m, 2)) = nda::arange(0, (m - 1) / 2);
358 piv(nda::range(2, m, 2)) = nda::arange(m - 1, (m - 1) / 2, -1);
359 //piv(m - 1) = (m - 1) / 2;
360 }
361
362 // If m odd, first choose middle row (now last row) as first pivot
363
364 if (m % 2 == 1) {
365 // Normalize
366 aa(0, _) /= sqrt(normsq(aa(0, _)));
367
368 // Orthogonalize remaining rows against current row
369 for (int k = 1; k < m; ++k) {
370 aa(k, _) = aa(k, _) - aa(0, _) * nda::blas::dotc(aa(0, _), aa(k, _));
371 norms(k) = normsq(aa(k, _));
372 }
373 }
374
375 int jstrt = (m % 2 == 1) ? 1 : 0;
376 // Then proceed with pivoted GS algorithm as normal
377 for (int j = jstrt; j < r; j += 2) {
378
379 // Find next pair of pivots
380 jpiv = j;
381 nrm = norms(j) + norms(j + 1);
382 for (int k = j + 2; k < m; k += 2) {
383 if (norms(k) + norms(k + 1) > nrm) {
384 jpiv = k;
385 nrm = norms(k) + norms(k + 1);
386 }
387 }
388
389 // Swap current row pair with chosen pivot row pair
390 deep_swap(aa(j, _), aa(jpiv, _));
391 deep_swap(aa(j + 1, _), aa(jpiv + 1, _));
392 std::swap(norms(j), norms(jpiv));
393 std::swap(norms(j + 1), norms(jpiv + 1));
394 std::swap(piv(j), piv(jpiv));
395 std::swap(piv(j + 1), piv(jpiv + 1));
396
397 // Orthogonalize current row (now the first chosen pivot row) against all
398 // previously chosen rows
399 for (int k = 0; k < j; ++k) { aa(j, _) = aa(j, _) - aa(k, _) * nda::blas::dotc(aa(k, _), aa(j, _)); }
400
401 // Normalize current row
402 aa(j, _) /= sqrt(normsq(aa(j, _)));
403
404 // Orthogonalize remaining rows against current row
405 for (int k = j + 1; k < m; ++k) {
406 aa(k, _) = aa(k, _) - aa(j, _) * nda::blas::dotc(aa(j, _), aa(k, _));
407 norms(k) = normsq(aa(k, _));
408 }
409
410 // Orthogonalize current row (now the second chosen pivot row) against all
411 // previously chosen rows
412 for (int k = 0; k < j + 1; ++k) { aa(j + 1, _) = aa(j + 1, _) - aa(k, _) * nda::blas::dotc(aa(k, _), aa(j + 1, _)); }
413
414 // Normalize current row
415 aa(j + 1, _) /= sqrt(normsq(aa(j + 1, _)));
416
417 // Orthogonalize remaining rows against current row
418 for (int k = j + 2; k < m; ++k) {
419 aa(k, _) = aa(k, _) - aa(j + 1, _) * nda::blas::dotc(aa(j + 1, _), aa(k, _));
420 norms(k) = normsq(aa(k, _));
421 }
422 }

424 return {aa(nda::range(r), _), norms(nda::range(r)), piv(nda::range(r))};
425 }

514 template <nda::MemoryArray Ta, nda::MemoryArray Tb, nda::Scalar Sa = nda::get_value_t<Ta>, nda::Scalar Sb = nda::get_value_t<Tb>,
515 nda::Scalar S = std::common_type_t<Sa, Sb>>
516 nda::array<S, Ta::rank + Tb::rank - 2> arraymult(Ta const &a, Tb const &b) {
517
518 // Get ranks of input arrays
519 constexpr int ra = Ta::rank;
520 constexpr int rb = Tb::rank;
521
522 // Get inner dimensions of input arrays
523 int p = a.shape(ra - 1);
524 if (b.shape(0) != p) throw std::runtime_error("last dim of a != first dim of b");
525
526 // Get product of outer dimensions of input arrays
527 int m = a.size() / p;
528 int n = b.size() / p;
529
530 // Reshape input arrays to 2D arrays
531 auto a_reshaped = nda::reshape(a, m, p);
532 auto b_reshaped = nda::reshape(b, p, n);
533
534 // Get shape of output array
535 auto c_shape = std::array<int, ra + rb - 2>();
536 for (int i = 0; i < ra - 1; ++i) { c_shape[i] = a.shape(i); }
537 for (int i = ra - 1; i < ra + rb - 2; ++i) { c_shape[i] = b.shape(i - ra + 2); }
538
539 // Compute the contraction, reshape, and return
540 return reshape(matmul(a_reshaped, b_reshaped), c_shape);
541 }

566 S adapgl(std::function<nda::array<S, 1>(nda::array<double, 1>)> f, double a, double b, double tol, nda::vector<double> xgl,
567 nda::vector<double> wgl) {
568
569 int maxnde = 100000; // Max nodes in integration tree
570 int maxstk = 1000; // Max stack size
571
572 S sum = 0; // Integral
573
574 // Initialize stack: there is a stack of interval endpoints associated w/
575 // each node in the integration tree, and a corresponding stack of values
576 // associated w/ each such interval
577
578 auto endpt = nda::array<double, 2>(maxstk, 2); // Stack of interval endpoints
579 auto sums = nda::array<S, 1>(maxstk); // Stack of sums associated w/ each interval
580
581 int istk = 0; // Stack pointer
582 int nnde = 1; // Number of nodes in integration tree
583
584 endpt(istk, 0) = a; // Root node interval is [a,b]
585 endpt(istk, 1) = b;
586
587 double h = (b - a) / 2; // Interval half-width
588 sums(istk) = h * nda::blas::dotc(wgl, f(xgl * h + (a + b) / 2)); // Integral of f on root node interval
589
590 double aa = 0, bb = 0, cc = 0; // Interval endpoints, midpoint
591 S sum1 = 0, sum2 = 0; // Integrals on subintervals
592
593 // Main loop
594 while (true) {
595
596 aa = endpt(istk, 0); // Get interval endpoints
597 bb = endpt(istk, 1);
598 cc = (aa + bb) / 2; // Get midpoint
599
600 // Get integral on each subinterval
601 h = (cc - aa) / 2;
602 sum1 = h * nda::blas::dotc(wgl, f(xgl * h + (aa + cc) / 2));
603 sum2 = h * nda::blas::dotc(wgl, f(xgl * h + (cc + bb) / 2));
604
605 // If error on subintervals is less than tolerance, add to integral and
606 // move to next node
607
608 if (std::abs(sum1 + sum2 - sums(istk)) < tol) {
609 sum += sum1 + sum2; // Add to subintervals to integral
610 istk--; // Remove current interval from stack
611 if (istk < 0) { return sum; } // If stack is empty, return integral
612 } else { // Integral on current interval not converged
613
614 // Remove current interval from stack, and add subintervals
615 endpt(istk, 1) = cc;
616 endpt(istk + 1, 0) = cc;
617 endpt(istk + 1, 1) = bb;
618 sums(istk) = sum1;
619 sums(istk + 1) = sum2;
620
621 istk++;
622 nnde += 2;
623
624 if (istk > maxstk) { throw std::runtime_error("integration stack too large"); }
625 if (nnde > maxnde) { throw std::runtime_error("integration tree too large"); }
626 }
627 }
