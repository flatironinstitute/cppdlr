111 template <nda::MemoryArrayOfRank<2> T, nda::Scalar S = nda::get_value_t<T>>
112 std::tuple<typename T::regular_type, nda::vector<double>, nda::vector<int>> pivrgs(T const &a, double eps) {
113
114 auto _ = nda::range::all;
115
116 // Copy input data
117 auto aa = make_regular(a);
118
119 // Get matrix dimensions
120 auto [m, n] = aa.shape();
121 int maxrnk = std::min(m, n);
122
123 // Compute norms of rows of input matrix, and rescale eps tolerance
124 auto norms = nda::vector<double>(m);
125 double epssq = eps * eps;
126 for (int j = 0; j < m; ++j) { norms(j) = normsq(aa(j, _)); }
127
128 // Begin pivoted double Gram-Schmidt procedure
129 int jpiv = 0;
130 double nrm = 0;
131 auto piv = nda::arange(m);
132 auto tmp = nda::vector<S>(n);
133
134 for (int j = 0; j < maxrnk; ++j) {
135
136 // Find next pivot
137 jpiv = j;
138 nrm = norms(j);
139 for (int k = j + 1; k < m; ++k) {
140 if (norms(k) > nrm) {
141 jpiv = k;
142 nrm = norms(k);
143 }
144 }
145
146 // Swap current row with chosen pivot row
147 deep_swap(aa(j, _), aa(jpiv, _));
148 std::swap(norms(j), norms(jpiv));
149 std::swap(piv(j), piv(jpiv));
150
151 // Orthogonalize current rows (now the chosen pivot row) against all
152 // previously chosen rows
153 for (int k = 0; k < j; ++k) { aa(j, _) = aa(j, _) - aa(k, _) * nda::blas::dotc(aa(k, _), aa(j, _)); }
154
155 // Get norm of current row
156 nrm = normsq(aa(j, _));
157
158 // Terminate if sufficiently small, and return previously selected rows
159 // (not including current row)
160 if (nrm <= epssq) { return {aa(nda::range(0, j), _), norms(nda::range(0, j)), piv(nda::range(0, j))}; };
161
162 // Normalize current row
163 aa(j, _) /= sqrt(nrm);
164
165 // Orthogonalize remaining rows against current row
166 for (int k = j + 1; k < m; ++k) {
167 if (norms(k) <= epssq) { continue; } // Can skip rows with norm less than tolerance
168 aa(k, _) = aa(k, _) - aa(j, _) * nda::blas::dotc(aa(j, _), aa(k, _));
169 norms(k) = normsq(aa(k, _));
170 }
171 }
172
173 return {aa(nda::range(maxrnk), _), norms(nda::range(maxrnk)), piv(nda::range(maxrnk))};
174 }

335 aa(nda::range(1, m, 2), _) = a(nda::range(m - 1, m / 2 - 1, -1), _);
336 } else {
337 aa(0, _) = a((m - 1) / 2, _);
338 aa(nda::range(1, m, 2), _) = a(nda::range(0, (m - 1) / 2), _);

353 piv(nda::range(1, m, 2)) = nda::arange(m - 1, m / 2 - 1, -1);

513 template <nda::MemoryArray Ta, nda::MemoryArray Tb, nda::Scalar Sa = nda::get_value_t<Ta>, nda::Scalar Sb = nda::get_value_t<Tb>,
514 nda::Scalar S = std::common_type_t<Sa, Sb>>
515 nda::array<S, Ta::rank + Tb::rank - 2> arraymult(Ta const &a, Tb const &b) {
516
517 // Get ranks of input arrays
518 constexpr int ra = Ta::rank;
519 constexpr int rb = Tb::rank;
520
521 // Get inner dimensions of input arrays
522 int p = a.shape(ra - 1);
523 if (b.shape(0) != p) throw std::runtime_error("last dim of a != first dim of b");
524
525 // Get product of outer dimensions of input arrays
526 int m = a.size() / p;
527 int n = b.size() / p;
528
529 // Reshape input arrays to 2D arrays
530 auto a_reshaped = nda::reshape(a, m, p);
531 auto b_reshaped = nda::reshape(b, p, n);
532
533 // Get shape of output array
534 auto c_shape = std::array<int, ra + rb - 2>();
535 for (int i = 0; i < ra - 1; ++i) { c_shape[i] = a.shape(i); }
536 for (int i = ra - 1; i < ra + rb - 2; ++i) { c_shape[i] = b.shape(i - ra + 2); }
537
538 // Compute the contraction, reshape, and return
539 return reshape(matmul(a_reshaped, b_reshaped), c_shape);
540 }

565 S adapgl(std::function<nda::array<S, 1>(nda::array<double, 1>)> f, double a, double b, double tol, nda::vector<double> xgl,
566 nda::vector<double> wgl) {
567
568 int maxnde = 100000; // Max nodes in integration tree
569 int maxstk = 1000; // Max stack size
570
571 S sum = 0; // Integral
572
573 // Initialize stack: there is a stack of interval endpoints associated w/
574 // each node in the integration tree, and a corresponding stack of values
575 // associated w/ each such interval
576
577 auto endpt = nda::array<double, 2>(maxstk, 2); // Stack of interval endpoints
578 auto sums = nda::array<S, 1>(maxstk); // Stack of sums associated w/ each interval
579
580 int istk = 0; // Stack pointer
581 int nnde = 1; // Number of nodes in integration tree
582
583 endpt(istk, 0) = a; // Root node interval is [a,b]
584 endpt(istk, 1) = b;
585
586 double h = (b - a) / 2; // Interval half-width
587 sums(istk) = h * nda::blas::dotc(wgl, f(xgl * h + (a + b) / 2)); // Integral of f on root node interval
588
589 double aa = 0, bb = 0, cc = 0; // Interval endpoints, midpoint
590 S sum1 = 0, sum2 = 0; // Integrals on subintervals
591
592 // Main loop
593 while (true) {
594
595 aa = endpt(istk, 0); // Get interval endpoints
596 bb = endpt(istk, 1);
597 cc = (aa + bb) / 2; // Get midpoint
598
599 // Get integral on each subinterval
600 h = (cc - aa) / 2;
601 sum1 = h * nda::blas::dotc(wgl, f(xgl * h + (aa + cc) / 2));
602 sum2 = h * nda::blas::dotc(wgl, f(xgl * h + (cc + bb) / 2));
603
604 // If error on subintervals is less than tolerance, add to integral and
605 // move to next node
606
607 if (std::abs(sum1 + sum2 - sums(istk)) < tol) {
608 sum += sum1 + sum2; // Add to subintervals to integral
609 istk--; // Remove current interval from stack
610 if (istk < 0) { return sum; } // If stack is empty, return integral
611 } else { // Integral on current interval not converged
612
613 // Remove current interval from stack, and add subintervals
614 endpt(istk, 1) = cc;
615 endpt(istk + 1, 0) = cc;
616 endpt(istk + 1, 1) = bb;
617 sums(istk) = sum1;
618 sums(istk + 1) = sum2;
619
620 istk++;
621 nnde += 2;
622
623 if (istk > maxstk) { throw std::runtime_error("integration stack too large"); }
624 if (nnde > maxnde) { throw std::runtime_error("integration tree too large"); }
625 }
626 }
