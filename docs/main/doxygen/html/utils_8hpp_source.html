116 template <nda::MemoryArrayOfRank<2> T, nda::Scalar S = nda::get_value_t<T>>
117 std::tuple<typename T::regular_type, nda::vector<double>, nda::vector<int>> pivrgs(T const &a, double eps) {
118
119 // Copy input data
120 auto aa = make_regular(a);
121
122 // Get matrix dimensions
123 auto [m, n] = aa.shape();
124 int maxrnk = std::min(m, n);
125
126 // Compute norms of rows of input matrix, and rescale eps tolerance
127 auto norms = nda::vector<double>(m);
128 double epssq = eps * eps;
129 for (int j = 0; j < m; ++j) { norms(j) = normsq(aa(j, _)); }
130
131 // Begin pivoted double Gram-Schmidt procedure
132 int jpiv = 0;
133 double nrm = 0;
134 auto piv = nda::arange(m);
135 auto tmp = nda::vector<S>(n);
136
137 for (int j = 0; j < maxrnk; ++j) {
138
139 // Find next pivot
140 jpiv = j;
141 nrm = norms(j);
142 for (int k = j + 1; k < m; ++k) {
143 if (norms(k) > nrm) {
144 jpiv = k;
145 nrm = norms(k);
146 }
147 }
148
149 // Swap current row with chosen pivot row
150 deep_swap(aa(j, _), aa(jpiv, _));
151 std::swap(norms(j), norms(jpiv));
152 std::swap(piv(j), piv(jpiv));
153
154 // Orthogonalize current rows (now the chosen pivot row) against all
155 // previously chosen rows
156 for (int k = 0; k < j; ++k) { aa(j, _) = aa(j, _) - aa(k, _) * nda::blas::dotc(aa(k, _), aa(j, _)); }
157
158 // Get norm of current row
159 nrm = normsq(aa(j, _));
160
161 // Terminate if sufficiently small, and return previously selected rows
162 // (not including current row)
163 if (nrm <= epssq) { return {aa(nda::range(0, j), _), norms(nda::range(0, j)), piv(nda::range(0, j))}; };
164
165 // Normalize current row
166 aa(j, _) /= sqrt(nrm);
167
168 // Orthogonalize remaining rows against current row
169 for (int k = j + 1; k < m; ++k) {
170 if (norms(k) <= epssq) { continue; } // Can skip rows with norm less than tolerance
171 aa(k, _) = aa(k, _) - aa(j, _) * nda::blas::dotc(aa(j, _), aa(k, _));
172 norms(k) = normsq(aa(k, _));
173 }
174 }
175
176 return {aa(nda::range(maxrnk), _), norms(nda::range(maxrnk)), piv(nda::range(maxrnk))};
177 }

206 // Get matrix dimensions

209
210 if (m % 2 != 0) { throw std::runtime_error("Input matrix must have even number of rows."); }
211
212 // Copy input data, re-ordering rows to make symmetric rows adjacent.
213 auto aa = typename T::regular_type(m, n);
214 aa(nda::range(0, m, 2), _) = a(nda::range(0, m / 2), _);
215 aa(nda::range(1, m, 2), _) = a(nda::range(m - 1, m / 2 - 1, -1), _);
216
217 // Compute norms of rows of input matrix, and rescale eps tolerance
218 auto norms = nda::vector<double>(m);
219 double epssq = eps * eps;
220 for (int j = 0; j < m; ++j) { norms(j) = normsq(aa(j, _)); }
221
222 // Begin pivoted double Gram-Schmidt procedure
223 int jpiv = 0;
224 double nrm = 0;
225 auto piv = nda::arange(0, m);
226 piv(nda::range(0, m, 2)) = nda::arange(0, m / 2); // Re-order pivots to match re-ordered input matrix
227 piv(nda::range(1, m, 2)) = nda::arange(m - 1, m / 2 - 1, -1);
228
229 if (maxrnk % 2 != 0) { // If n < m and n is odd, decrease maxrnk to maintain symmetry
230 maxrnk -= 1;
231 }
232
233 for (int j = 0; j < maxrnk; j += 2) {
234
235 // Find next pair of pivots
236 jpiv = j;
237 nrm = norms(j) + norms(j + 1);
238 for (int k = j + 2; k < m; k += 2) {
239 if (norms(k) + norms(k + 1) > nrm) {
240 jpiv = k;
241 nrm = norms(k) + norms(k + 1);
242 }
243 }
244
245 // Swap current row pair with chosen pivot row pair
246 deep_swap(aa(j, _), aa(jpiv, _));
247 deep_swap(aa(j + 1, _), aa(jpiv + 1, _));
248 std::swap(norms(j), norms(jpiv));
249 std::swap(norms(j + 1), norms(jpiv + 1));
250 std::swap(piv(j), piv(jpiv));
251 std::swap(piv(j + 1), piv(jpiv + 1));
252
253 // Orthogonalize current row (now the first chosen pivot row) against all
254 // previously chosen rows
255 for (int k = 0; k < j; ++k) { aa(j, _) = aa(j, _) - aa(k, _) * nda::blas::dotc(aa(k, _), aa(j, _)); }
256
257 // Get norm of current row
258 nrm = normsq(aa(j, _));
259
260 // Terminate if sufficiently small, and return previously selected rows
261 // (not including current row)
262 if (nrm <= epssq) { return {aa(nda::range(0, j), _), norms(nda::range(0, j)), piv(nda::range(0, j))}; };
263
264 // Normalize current row
265 aa(j, _) /= sqrt(nrm);
266
267 // Orthogonalize remaining rows against current row
268 for (int k = j + 1; k < m; ++k) {
269 if (norms(k) <= epssq) { continue; } // Can skip rows with norm less than tolerance
270 aa(k, _) = aa(k, _) - aa(j, _) * nda::blas::dotc(aa(j, _), aa(k, _));
271 norms(k) = normsq(aa(k, _));
272 }
273
274 // Orthogonalize current row (now the second chosen pivot row) against all
275 // previously chosen rows
276 for (int k = 0; k < j + 1; ++k) { aa(j + 1, _) = aa(j + 1, _) - aa(k, _) * nda::blas::dotc(aa(k, _), aa(j + 1, _)); }
277
278 // Normalize current row
279 aa(j + 1, _) /= sqrt(normsq(aa(j + 1, _)));
280
281 // Orthogonalize remaining rows against current row
282 for (int k = j + 2; k < m; ++k) {
283 if (norms(k) <= epssq) { continue; } // Can skip rows with norm less than tolerance
284 aa(k, _) = aa(k, _) - aa(j + 1, _) * nda::blas::dotc(aa(j + 1, _), aa(k, _));
285 norms(k) = normsq(aa(k, _));
286 }
287 }
288
289 return {aa(nda::range(maxrnk), _), norms(nda::range(maxrnk)), piv(nda::range(maxrnk))};
290 }

319 std::tuple<typename T::regular_type, nda::vector<double>, nda::vector<int>> pivrgs_sym(T const &a, int r) {
320
321 // Get matrix dimensions
322 auto [m, n] = a.shape();
323
324 if (m % 2 == 1 && r % 2 == 0) { throw std::runtime_error("If input matrix has odd number of rows, r must be odd."); }
325 if (r % 2 == 1 && m % 2 == 0) { throw std::runtime_error("If r is odd, input matrix must have odd number of rows."); }
326 if (r > m || r > n + 1) { throw std::runtime_error("r must be less than or equal to min(m,n+1)."); }
327 if (r == n + 1 && (n % 2 == 1 || n > m)) { throw std::runtime_error("If r = n+1, n must be even and less than or equal to m."); }
328
329 // Copy input data, re-ordering rows to make symmetric rows adjacent. If m
330 // odd, put middle row at the end.
331 auto aa = typename T::regular_type(m, n);
332 if (m % 2 == 0) {
333 aa(nda::range(0, m, 2), _) = a(nda::range(0, m / 2), _);
334 aa(nda::range(1, m, 2), _) = a(nda::range(m - 1, m / 2 - 1, -1), _);
335 } else {
336 aa(0, _) = a((m - 1) / 2, _);
337 aa(nda::range(1, m, 2), _) = a(nda::range(0, (m - 1) / 2), _);
338 aa(nda::range(2, m, 2), _) = a(nda::range(m - 1, (m - 1) / 2, -1), _);
339 //aa(m - 1, _) = a((m - 1) / 2, _);
340 }
341
342 // Compute norms of rows of input matrix
343 auto norms = nda::vector<double>(m);
344 for (int j = 0; j < m; ++j) { norms(j) = normsq(aa(j, _)); }
345
346 // Begin pivoted double Gram-Schmidt procedure
347 int jpiv = 0;
348 double nrm = 0;
349 auto piv = nda::arange(0, m);
350 if (m % 2 == 0) {
351 piv(nda::range(0, m, 2)) = nda::arange(0, m / 2); // Re-order pivots to match re-ordered input matrix
352 piv(nda::range(1, m, 2)) = nda::arange(m - 1, m / 2 - 1, -1);
353 } else {
354 piv(0) = (m - 1) / 2;
355 piv(nda::range(1, m, 2)) = nda::arange(0, (m - 1) / 2);
356 piv(nda::range(2, m, 2)) = nda::arange(m - 1, (m - 1) / 2, -1);
357 //piv(m - 1) = (m - 1) / 2;
358 }
359
360 // If m odd, first choose middle row (now last row) as first pivot
361
362 if (m % 2 == 1) {
363 // Normalize
364 aa(0, _) /= sqrt(normsq(aa(0, _)));
365
366 // Orthogonalize remaining rows against current row
367 for (int k = 1; k < m; ++k) {
368 aa(k, _) = aa(k, _) - aa(0, _) * nda::blas::dotc(aa(0, _), aa(k, _));
369 norms(k) = normsq(aa(k, _));
370 }
371 }
372
373 int jstrt = (m % 2 == 1) ? 1 : 0;
374 // Then proceed with pivoted GS algorithm as normal
375 for (int j = jstrt; j < r; j += 2) {
376
377 // Find next pair of pivots
378 jpiv = j;
379 nrm = norms(j) + norms(j + 1);
380 for (int k = j + 2; k < m; k += 2) {
381 if (norms(k) + norms(k + 1) > nrm) {
382 jpiv = k;
383 nrm = norms(k) + norms(k + 1);
384 }
385 }
386
387 // Swap current row pair with chosen pivot row pair
388 deep_swap(aa(j, _), aa(jpiv, _));
389 deep_swap(aa(j + 1, _), aa(jpiv + 1, _));
390 std::swap(norms(j), norms(jpiv));
391 std::swap(norms(j + 1), norms(jpiv + 1));
392 std::swap(piv(j), piv(jpiv));
393 std::swap(piv(j + 1), piv(jpiv + 1));
394
395 // Orthogonalize current row (now the first chosen pivot row) against all
396 // previously chosen rows
397 for (int k = 0; k < j; ++k) { aa(j, _) = aa(j, _) - aa(k, _) * nda::blas::dotc(aa(k, _), aa(j, _)); }
398
399 // Normalize current row
400 aa(j, _) /= sqrt(normsq(aa(j, _)));
401
402 // Orthogonalize remaining rows against current row
403 for (int k = j + 1; k < m; ++k) {
404 aa(k, _) = aa(k, _) - aa(j, _) * nda::blas::dotc(aa(j, _), aa(k, _));
405 norms(k) = normsq(aa(k, _));
406 }
407
408 // Orthogonalize current row (now the second chosen pivot row) against all
409 // previously chosen rows
410 for (int k = 0; k < j + 1; ++k) { aa(j + 1, _) = aa(j + 1, _) - aa(k, _) * nda::blas::dotc(aa(k, _), aa(j + 1, _)); }
411
412 // Normalize current row
413 aa(j + 1, _) /= sqrt(normsq(aa(j + 1, _)));
414
415 // Orthogonalize remaining rows against current row
416 for (int k = j + 2; k < m; ++k) {
417 aa(k, _) = aa(k, _) - aa(j + 1, _) * nda::blas::dotc(aa(j + 1, _), aa(k, _));
418 norms(k) = normsq(aa(k, _));
419 }
420 }

423 }

512 template <nda::MemoryArray Ta, nda::MemoryArray Tb, nda::Scalar Sa = nda::get_value_t<Ta>, nda::Scalar Sb = nda::get_value_t<Tb>,
513 nda::Scalar S = std::common_type_t<Sa, Sb>>
514 nda::array<S, Ta::rank + Tb::rank - 2> arraymult(Ta const &a, Tb const &b) {
515
516 // Get ranks of input arrays
517 constexpr int ra = Ta::rank;
518 constexpr int rb = Tb::rank;
519
520 // Get inner dimensions of input arrays
521 int p = a.shape(ra - 1);
522 if (b.shape(0) != p) throw std::runtime_error("last dim of a != first dim of b");
523
524 // Get product of outer dimensions of input arrays
525 int m = a.size() / p;
526 int n = b.size() / p;
527
528 // Reshape input arrays to 2D arrays
529 auto a_reshaped = nda::reshape(a, m, p);
530 auto b_reshaped = nda::reshape(b, p, n);
531
532 // Get shape of output array
533 auto c_shape = std::array<int, ra + rb - 2>();
534 for (int i = 0; i < ra - 1; ++i) { c_shape[i] = a.shape(i); }
535 for (int i = ra - 1; i < ra + rb - 2; ++i) { c_shape[i] = b.shape(i - ra + 2); }
536
537 // Compute the contraction, reshape, and return
538 return reshape(matmul(a_reshaped, b_reshaped), c_shape);
539 }

564 S adapgl(std::function<nda::array<S, 1>(nda::array<double, 1>)> f, double a, double b, double tol, nda::vector<double> xgl,
565 nda::vector<double> wgl) {
566
567 int maxnde = 100000; // Max nodes in integration tree
568 int maxstk = 1000; // Max stack size
569
570 S sum = 0; // Integral
571
572 // Initialize stack: there is a stack of interval endpoints associated w/
573 // each node in the integration tree, and a corresponding stack of values
574 // associated w/ each such interval
575
576 auto endpt = nda::array<double, 2>(maxstk, 2); // Stack of interval endpoints
577 auto sums = nda::array<S, 1>(maxstk); // Stack of sums associated w/ each interval
578
579 int istk = 0; // Stack pointer
580 int nnde = 1; // Number of nodes in integration tree
581
582 endpt(istk, 0) = a; // Root node interval is [a,b]
583 endpt(istk, 1) = b;
584
585 double h = (b - a) / 2; // Interval half-width
586 sums(istk) = h * nda::blas::dotc(wgl, f(xgl * h + (a + b) / 2)); // Integral of f on root node interval
587
588 double aa = 0, bb = 0, cc = 0; // Interval endpoints, midpoint
589 S sum1 = 0, sum2 = 0; // Integrals on subintervals
590
591 // Main loop
592 while (true) {
593
594 aa = endpt(istk, 0); // Get interval endpoints
595 bb = endpt(istk, 1);
596 cc = (aa + bb) / 2; // Get midpoint
597
598 // Get integral on each subinterval
599 h = (cc - aa) / 2;
600 sum1 = h * nda::blas::dotc(wgl, f(xgl * h + (aa + cc) / 2));
601 sum2 = h * nda::blas::dotc(wgl, f(xgl * h + (cc + bb) / 2));
602
603 // If error on subintervals is less than tolerance, add to integral and
604 // move to next node
605
606 if (std::abs(sum1 + sum2 - sums(istk)) < tol) {
607 sum += sum1 + sum2; // Add to subintervals to integral
608 istk--; // Remove current interval from stack
609 if (istk < 0) { return sum; } // If stack is empty, return integral
610 } else { // Integral on current interval not converged
611
612 // Remove current interval from stack, and add subintervals
613 endpt(istk, 1) = cc;
614 endpt(istk + 1, 0) = cc;
615 endpt(istk + 1, 1) = bb;
616 sums(istk) = sum1;
617 sums(istk + 1) = sum2;
618
619 istk++;
620 nnde += 2;
621
622 if (istk > maxstk) { throw std::runtime_error("integration stack too large"); }
623 if (nnde > maxnde) { throw std::runtime_error("integration tree too large"); }
624 }
625 }
