103 // function, which is of type Tsig, with scalar type replaced by the common

106 template <nda::MemoryArray Tsig, nda::MemoryArray Tg = make_common_t<Tsig, Sh, nda::get_value_t<Tsig>>> Tg solve(Tsig const &sig) const {
107
108 int r = itops_ptr->rank(); // DLR rank
109 auto sigc = itops_ptr->vals2coefs(sig); // DLR coefficients of self-energy
110
111 // Obtain Dyson equation system matrix I - G0 * Sig, where G0 and Sig are the
112 // matrices of convolution by the free Green's function and self-energy,
113 // respectively.
114 auto sysmat = make_regular(nda::eye<double>(r * norb) - g0mat * itops_ptr->convmat(beta, Fermion, sigc, time_order));
115
116 // Factorize system matrix
117 auto ipiv = nda::vector<int>(r * norb);
118 nda::lapack::getrf(sysmat, ipiv);
119
120 // Solve Dyson equation
121 auto g = Tg(sig.shape()); // Declare Green's function
122 g = rhs; // Get right hand side of Dyson equation
123 auto g_rs = nda::matrix_view<nda::get_value_t<Tg>>(nda::reshape(g, norb, r * norb)); // Reshape g to be compatible w/ LAPACK
124 nda::lapack::getrs(sysmat, g_rs, ipiv); // Back solve
125 if constexpr (std::floating_point<Ht>) { // If h is scalar, g is scalar-valued
126 return g;
127 } else { // Otherwise, g is matrix-valued, and need to transpose some indices after solve to undo LAPACK formatting
128 return permuted_indices_view<nda::encode<3>({2, 0, 1})>(g);
129 }
130 }

165 // If h is scalar, return scalar-valued Green's function; if h is matrix,
166 // return matrix-valued Green's function
167 auto free_gf(double beta, imtime_ops const &itops, Ht const &h, double mu = 0, bool time_order = false) {
168
169 int r = itops.rank();
170
171 if constexpr (std::floating_point<Ht>) { // If h is scalar, return scalar-valued Green's function
172 auto g = nda::array<Ht, 1>(r);
173 if (!time_order) { // Free Green's function for ordinary Dyson equation
174 for (int i = 0; i < r; i++) { g(i) = k_it(itops.get_itnodes(i), h - mu, beta); }
175 } else { // Free Green's function for time-ordered Dyson equation
176 for (int i = 0; i < r; i++) { g(i) = -exp(-beta * itops.get_itnodes(i) * (h - mu)); }
177 }
178 return g;
179 } else { // Otherwise, return matrix-valued Green's function
180
181 int norb = h.shape(0);
182
183 // Diagonalize Hamiltonian
184 auto [eval, evec] = nda::linalg::eigenelements(h);
185
186 // Get free Green's function
187 auto g = nda::array<nda::get_value_t<Ht>, 3>(r, norb, norb);
188 g = 0;
189 for (int i = 0; i < r; i++) {
190 if (!time_order) {
191 for (int j = 0; j < norb; j++) { g(i, j, j) = k_it(itops.get_itnodes(i), eval(j) - mu, beta); }
192 } else {
193 for (int j = 0; j < norb; j++) { g(i, j, j) = -exp(-beta * rel2abs(itops.get_itnodes(i)) * (eval(j) - mu)); }
194 }
195 g(i, _, _) = matmul(evec, matmul(g(i, _, _), transpose(conj(evec))));
196 }
197
198 return g;
199 }
200 }
